142 static void _twr_cmwx1zzabz_task(void *param)
143 {
144  twr_cmwx1zzabz_t *self = param;
145 
146  while (true)
147  {
148  switch (self->_state)
149  {
150  case TWR_CMWX1ZZABZ_STATE_READY:
151  {
152  self->_state = TWR_CMWX1ZZABZ_STATE_IDLE;
153 
154  if (self->_event_handler != NULL)
155  {
156  self->_event_handler(self, TWR_CMWX1ZZABZ_EVENT_READY, self->_event_param);
157  }
158 
159  continue;
160  }
161  case TWR_CMWX1ZZABZ_STATE_IDLE:
162  {
163  if (self->_save_config_mask != 0)
164  {
165  self->_state = TWR_CMWX1ZZABZ_STATE_CONFIG_SAVE_SEND;
166  continue;
167  }
168 
169  if(self->_join_command)
170  {
171  self->_state = TWR_CMWX1ZZABZ_STATE_JOIN_SEND;
172  continue;
173  }
174 
175  return;
176  }
177  case TWR_CMWX1ZZABZ_STATE_RECEIVE:
178  {
179  self->_state = TWR_CMWX1ZZABZ_STATE_IDLE;
180 
181  while (_twr_cmwx1zzabz_read_response(self))
182  {
183  if (memcmp(self->_response, "+RECV=", 5) == 0)
184  {
185  self->_message_port = atoi(&self->_response[6]);
186 
187  char *comma_search = strchr(self->_response, ',');
188  if (!comma_search)
189  {
190  continue;
191  }
192 
193  // Parse from the next character
194  self->_message_length = atoi(++comma_search);
195 
196  // Dummy read three \r\n\r characters
197  char dummy[3];
198  uint32_t bytes = twr_uart_async_read(self->_uart_channel, &dummy, 3);
199  if (bytes != 3)
200  {
201  continue;
202  }
203 
204  // Received data is bigger than library message buffer
205  if (self->_message_length > sizeof(self->_message_buffer))
206  {
207  continue;
208  }
209 
210  // Read the received message
211  bytes = twr_uart_async_read(self->_uart_channel, self->_message_buffer, self->_message_length);
212  if (bytes != self->_message_length)
213  {
214  continue;
215  }
216 
217  if (self->_event_handler != NULL)
218  {
219  self->_event_handler(self, TWR_CMWX1ZZABZ_EVENT_MESSAGE_RECEIVED, self->_event_param);
220  }
221  }
222  else if (memcmp(self->_response, "+ACK", 4) == 0)
223  {
224  if (self->_event_handler != NULL)
225  {
226  self->_event_handler(self, TWR_CMWX1ZZABZ_EVENT_MESSAGE_CONFIRMED, self->_event_param);
227  }
228  }
229  else if (memcmp(self->_response, "+NOACK", 4) == 0)
230  {
231  if (self->_event_handler != NULL)
232  {
233  self->_event_handler(self, TWR_CMWX1ZZABZ_EVENT_MESSAGE_NOT_CONFIRMED, self->_event_param);
234  }
235  }
236  else if (memcmp(self->_response, "+EVENT=2,2", 10) == 0)
237  {
238  if (self->_event_handler != NULL)
239  {
240  self->_event_handler(self, TWR_CMWX1ZZABZ_EVENT_MESSAGE_RETRANSMISSION, self->_event_param);
241  }
242  }
243  }
244 
245  return;
246  }
247  case TWR_CMWX1ZZABZ_STATE_ERROR:
248  {
249  if (self->_event_handler != NULL)
250  {
251  self->_event_handler(self, TWR_CMWX1ZZABZ_EVENT_ERROR, self->_event_param);
252  }
253 
254  self->_state = TWR_CMWX1ZZABZ_STATE_INITIALIZE;
255 
256  continue;
257  }
258  case TWR_CMWX1ZZABZ_STATE_INITIALIZE:
259  {
260  self->_init_command_index = 0;
261  self->_state = TWR_CMWX1ZZABZ_STATE_INITIALIZE_COMMAND_SEND;
262 
263  continue;
264  }
265 
266  case TWR_CMWX1ZZABZ_STATE_INITIALIZE_COMMAND_SEND:
267  {
268  self->_state = TWR_CMWX1ZZABZ_STATE_ERROR;
269 
270  // Purge RX FIFO
271  char rx_character;
272  while (twr_uart_async_read(self->_uart_channel, &rx_character, 1) != 0)
273  {
274  }
275 
276  strcpy(self->_command, _init_commands[self->_init_command_index]);
277  size_t length = strlen(self->_command);
278 
279  if (_twr_cmwx1zzabz_async_write(self, self->_uart_channel, self->_command, length) != length)
280  {
281  continue;
282  }
283 
284  self->_state = TWR_CMWX1ZZABZ_STATE_INITIALIZE_COMMAND_RESPONSE;
285  twr_scheduler_plan_current_from_now(TWR_CMWX1ZZABZ_DELAY_INITIALIZATION_AT_COMMAND);
286 
287  return;
288  }
289  case TWR_CMWX1ZZABZ_STATE_INITIALIZE_COMMAND_RESPONSE:
290  {
291  self->_state = TWR_CMWX1ZZABZ_STATE_ERROR;
292  uint8_t response_handled = 0;
293 
294  if (!_twr_cmwx1zzabz_read_response(self))
295  {
296  continue;
297  }
298 
299  // Compare first 4 cahracters from response
300  uint32_t response_valid = (memcmp(self->_response, "+OK=", 4) == 0);
301  // Pointer to the last send command to know the context of the answer
302  const char *last_command = _init_commands[self->_init_command_index];
303  // Pointer to the first character of response value after +OK=
304  char *response_string_value = &self->_response[4];
305 
306  if (strcmp(last_command, "AT+DEVADDR?\r") == 0 && response_valid)
307  {
308  // Check if user did not filled this structure to save configuration, oterwise it would be overwritten
309  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_DEVADDR) == 0)
310  {
311  memcpy(self->_config.devaddr, response_string_value, 8);
312  self->_config.devaddr[8] = '\0';
313  }
314  response_handled = 1;
315  }
316  else if (strcmp(last_command, "AT+DEVEUI?\r") == 0 && response_valid)
317  {
318  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_DEVEUI) == 0)
319  {
320  memcpy(self->_config.deveui, response_string_value, 16);
321  self->_config.deveui[16] = '\0';
322  }
323  response_handled = 1;
324  }
325  else if (strcmp(last_command, "AT+APPEUI?\r") == 0 && response_valid)
326  {
327  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_APPEUI) == 0)
328  {
329  memcpy(self->_config.appeui, response_string_value, 16);
330  self->_config.appeui[16] = '\0';
331  }
332  response_handled = 1;
333  }
334  else if (strcmp(last_command, "AT+NWKSKEY?\r") == 0 && response_valid)
335  {
336  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_NWKSKEY) == 0)
337  {
338  memcpy(self->_config.nwkskey, response_string_value, 32);
339  self->_config.nwkskey[32] = '\0';
340  }
341  response_handled = 1;
342  }
343  else if (strcmp(last_command, "AT+APPSKEY?\r") == 0 && response_valid)
344  {
345  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_APPSKEY) == 0)
346  {
347  memcpy(self->_config.appskey, response_string_value, 32);
348  self->_config.appskey[32] = '\0';
349  }
350  response_handled = 1;
351  }
352  else if (strcmp(last_command, "AT+APPKEY?\r") == 0 && response_valid)
353  {
354  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_APPKEY) == 0)
355  {
356  memcpy(self->_config.appkey, response_string_value, 32);
357  self->_config.appkey[32] = '\0';
358  }
359  response_handled = 1;
360  }
361  else if (strcmp(last_command, "AT+BAND?\r") == 0 && response_valid)
362  {
363  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_BAND) == 0)
364  {
365  self->_config.band = response_string_value[0] - '0';
366  }
367  response_handled = 1;
368  }
369  else if (strcmp(last_command, "AT+MODE?\r") == 0 && response_valid)

374  }
375  response_handled = 1;
376  }
377  else if (strcmp(last_command, "AT+CLASS?\r") == 0 && response_valid)
378  {
379  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_CLASS) == 0)
380  {
381  self->_config.class = response_string_value[0] - '0';
382  }
383  response_handled = 1;
384  }
385  else if (strcmp(last_command, "AT+RX2?\r") == 0 && response_valid)
386  {
387  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_RX2) == 0)
388  {
389  self->_config.rx2_frequency = atoi(response_string_value);
390 
391  char *comma_search = strchr(response_string_value, ',');
392  if (!comma_search)
393  {
394  continue;
395  }
396 
397  self->_config.rx2_datarate = atoi(++comma_search);
398  }
399  response_handled = 1;
400  }
401  else if (strcmp(last_command, "AT+NWK?\r") == 0 && response_valid)
402  {
403  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_NWK) == 0)
404  {
405  self->_config.nwk_public = response_string_value[0] - '0';
406  }
407  response_handled = 1;
408  }
409  else if (strcmp(last_command, "AT+ADR?\r") == 0 && response_valid)
410  {
411  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_ADAPTIVE_DATARATE) == 0)
412  {
413  self->_config.adaptive_datarate = response_string_value[0] == '1';
414  }
415  response_handled = 1;
416  }
417  else if (strcmp(last_command, "AT+DR?\r") == 0 && response_valid)
418  {
419  if ((self->_save_config_mask & 1 << TWR_CMWX1ZZABZ_CONFIG_INDEX_DATARATE) == 0)
420  {
421  self->_config.datarate = atoi(response_string_value);
422  }
423  response_handled = 1;
424  }
425  else if (strcmp(last_command, "AT+DUTYCYCLE=0\r") == 0 && strcmp(self->_response, "+ERR=-17\r") == 0)
426  {
427  // DUTYCYLE is unusable in some band configuration, ignore this err response
428  response_handled = 1;
429  }
430  else if ( strcmp(last_command, "AT\r") == 0 &&
431  strcmp(self->_response, "+OK\r") == 0
432  )
433  {
434  response_handled = 1;
435  }
436  // Generic OK response to other commands
437  else if (memcmp(self->_response, "+OK", 3) == 0)
438  {
439  response_handled = 1;
440  }
441 
442  if (!response_handled)
443  {
444  continue;
445  }
446 
447  self->_init_command_index++;
448 
449  if (_init_commands[self->_init_command_index] == NULL)
450  {
451  // If configuration was changed and flag set, save them
452  if (self->_save_config_mask)
453  {
454  self->_state = TWR_CMWX1ZZABZ_STATE_CONFIG_SAVE_SEND;
455  self->_save_command_index = 0;
456  }
457  else
458  {
459  self->_state = TWR_CMWX1ZZABZ_STATE_READY;
460  }
461  }
462  else
463  {
464  self->_state = TWR_CMWX1ZZABZ_STATE_INITIALIZE_COMMAND_SEND;
465  }
466 
467  continue;
468  }
469  case TWR_CMWX1ZZABZ_STATE_SEND_MESSAGE_COMMAND:
470  case TWR_CMWX1ZZABZ_STATE_SEND_MESSAGE_CONFIRMED_COMMAND:
471  {
472  if (self->_state == TWR_CMWX1ZZABZ_STATE_SEND_MESSAGE_CONFIRMED_COMMAND)
473  {
474  snprintf(self->_command, TWR_CMWX1ZZABZ_TX_FIFO_BUFFER_SIZE, "AT+PCTX %d,%d\r", self->_tx_port, self->_message_length);
475  }
476  else
477  {
478  snprintf(self->_command, TWR_CMWX1ZZABZ_TX_FIFO_BUFFER_SIZE, "AT+PUTX %d,%d\r", self->_tx_port, self->_message_length);
479  }
480 
481  self->_state = TWR_CMWX1ZZABZ_STATE_ERROR;
482 
483  uint8_t command_length = strlen(self->_command);
484 
485  for (size_t i = 0; i < self->_message_length; i++)
486  {
487  // put binary data directly to the "string" buffer
488  self->_command[command_length + i] = self->_message_buffer[i];
489  }
490 
491  self->_command[command_length + self->_message_length] = '\r';
492 
493  size_t length = command_length + self->_message_length + 1; // 1 for \n
494 
495  if (_twr_cmwx1zzabz_async_write(self, self->_uart_channel, self->_command, length) != length)
496  {
497  continue;
498  }
499 
500  self->_state = TWR_CMWX1ZZABZ_STATE_SEND_MESSAGE_RESPONSE;
501 
502  if (self->_event_handler != NULL)
503  {
504  self->_event_handler(self, TWR_CMWX1ZZABZ_EVENT_SEND_MESSAGE_START, self->_event_param);
505  }
506 
507  twr_scheduler_plan_current_from_now(TWR_CMWX1ZZABZ_DELAY_SEND_MESSAGE_RESPONSE);
508 
509  return;
510  }
511  case TWR_CMWX1ZZABZ_STATE_SEND_MESSAGE_RESPONSE:
512  {
513  self->_state = TWR_CMWX1ZZABZ_STATE_ERROR;
514 
515  if (!_twr_cmwx1zzabz_read_response(self))
516  {
517  continue;
518  }
519 
520  if (strcmp(self->_response, "+OK\r") != 0)
521  {
522  continue;
523  }
524 
525  self->_state = TWR_CMWX1ZZABZ_STATE_READY;
526 
527  if (self->_event_handler != NULL)
528  {
529  self->_event_handler(self, TWR_CMWX1ZZABZ_EVENT_SEND_MESSAGE_DONE, self->_event_param);
530  }
531 
532  continue;
533  }
534  case TWR_CMWX1ZZABZ_STATE_CONFIG_SAVE_SEND:
535  {
536  self->_state = TWR_CMWX1ZZABZ_STATE_ERROR;
537 
538  // There are no more config items to send
539  if (self->_save_config_mask == 0)
540  {
541  if (self->_event_handler != NULL)
542  {
543  self->_event_handler(self, TWR_CMWX1ZZABZ_EVENT_CONFIG_SAVE_DONE, self->_event_param);
544  }
545 
546  self->_state = TWR_CMWX1ZZABZ_STATE_READY;
547  continue;
548  }
549 
550  // Find config item that has been changed
551  for (uint8_t i = 0; i < TWR_CMWX1ZZABZ_CONFIG_INDEX_LAST_ITEM; i++)
552  {
553  if (self->_save_config_mask & 1 << i)
554  {
555  self->_save_command_index = i;
556  break;
557  }
558  }
559 
560  // Purge RX FIFO
561  char rx_character;
562  while (twr_uart_async_read(self->_uart_channel, &rx_character, 1) != 0)
563  {
564  }
565 
566  switch (self->_save_command_index)
567  {
568  case TWR_CMWX1ZZABZ_CONFIG_INDEX_DEVADDR:
569  {
570  snprintf(self->_command, TWR_CMWX1ZZABZ_TX_FIFO_BUFFER_SIZE, "AT+DEVADDR=%s\r", self->_config.devaddr);
571  break;
572  }
573  case TWR_CMWX1ZZABZ_CONFIG_INDEX_DEVEUI:
574  {
575  snprintf(self->_command, TWR_CMWX1ZZABZ_TX_FIFO_BUFFER_SIZE, "AT+DEVEUI=%s\r", self->_config.deveui);
576  break;
577  }
578  case TWR_CMWX1ZZABZ_CONFIG_INDEX_APPEUI:
579  {
580  snprintf(self->_command, TWR_CMWX1ZZABZ_TX_FIFO_BUFFER_SIZE, "AT+APPEUI=%s\r", self->_config.appeui);
581  break;
582  }
583  case TWR_CMWX1ZZABZ_CONFIG_INDEX_NWKSKEY:
584  {
585  snprintf(self->_command, TWR_CMWX1ZZABZ_TX_FIFO_BUFFER_SIZE, "AT+NWKSKEY=%s\r", self->_config.nwkskey);
586  break;
587  }
588  case TWR_CMWX1ZZABZ_CONFIG_INDEX_APPSKEY:
589  {
590  snprintf(self->_command, TWR_CMWX1ZZABZ_TX_FIFO_BUFFER_SIZE, "AT+APPSKEY=%s\r", self->_config.appskey);
591  break;
592  }
593  case TWR_CMWX1ZZABZ_CONFIG_INDEX_APPKEY:
594  {
595  snprintf(self->_command, TWR_CMWX1ZZABZ_TX_FIFO_BUFFER_SIZE, "AT+APPKEY=%s\r", self->_config.appkey);
596  break;
597  }
