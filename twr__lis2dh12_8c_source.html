119 static void _twr_lis2dh12_task_measure(void *param)
120 {
121  twr_lis2dh12_t *self = param;
122 
123  while (true)
124  {
125  switch (self->_state)
126  {
127  case TWR_LIS2DH12_STATE_ERROR:
128  {
129  self->_accelerometer_valid = false;
130 
131  self->_measurement_active = false;
132 
133  if (self->_event_handler != NULL)
134  {
135  self->_event_handler(self, TWR_LIS2DH12_EVENT_ERROR, self->_event_param);
136  }
137 
138  self->_state = TWR_LIS2DH12_STATE_INITIALIZE;
139 
140  return;
141  }
142  case TWR_LIS2DH12_STATE_INITIALIZE:
143  {
144  self->_state = TWR_LIS2DH12_STATE_ERROR;
145 
146  // Read and check WHO_AM_I register
147  uint8_t who_am_i;
148  if (!twr_i2c_memory_read_8b(self->_i2c_channel, self->_i2c_address, 0x0f, &who_am_i))
149  {
150  continue;
151  }
152 
153  if (who_am_i != 0x33)
154  {
155  continue;
156  }
157 
158  uint8_t cfg_reg4 = 0x80 | ((uint8_t) self->_scale << 4) | (((uint8_t) self->_resolution & 0x01) << 3);
159 
160  if (!twr_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, 0x23, cfg_reg4))
161  {
162  continue;
163  }
164 
165  if (!_twr_lis2dh12_power_down(self))
166  {
167  continue;
168  }
169 
170  self->_state = TWR_LIS2DH12_STATE_MEASURE;
171 
172  if (self->_measurement_active)
173  {
174  twr_scheduler_plan_current_from_now(10);
175  }
176 
177  return;
178  }
179  case TWR_LIS2DH12_STATE_MEASURE:
180  {
181  self->_state = TWR_LIS2DH12_STATE_ERROR;
182 
183  if (!_twr_lis2dh12_continuous_conversion(self))
184  {
185  continue;
186  }
187 
188  self->_state = TWR_LIS2DH12_STATE_READ;
189 
190  twr_scheduler_plan_current_from_now(_TWR_LIS2DH12_DELAY_READ);
191 
192  return;
193  }
194  case TWR_LIS2DH12_STATE_READ:
195  {
196  self->_state = TWR_LIS2DH12_STATE_ERROR;
197 
198  if (!_twr_lis2dh12_read_result(self))
199  {
200  continue;
201  }
202 
203  // Power down only when no alarm is set
204  if(!self->_alarm_active)
205  {
206  if (!_twr_lis2dh12_power_down(self))
207  {
208  continue;
209  }
210  }
211 
212  self->_accelerometer_valid = true;
213 
214  self->_state = TWR_LIS2DH12_STATE_UPDATE;
215 
216  continue;
217  }
218  case TWR_LIS2DH12_STATE_UPDATE:
219  {
220  self->_state = TWR_LIS2DH12_STATE_ERROR;
221 
222  self->_measurement_active = false;
223 
224  if (self->_event_handler != NULL)
225  {
226  self->_event_handler(self, TWR_LIS2DH12_EVENT_UPDATE, self->_event_param);
227  }
228 
229  // When the interrupt alarm is active
230  if(self->_alarm_active)
231  {
232  uint8_t int1_src;
233 
234  if(!twr_i2c_memory_read_8b(self->_i2c_channel, self->_i2c_address, 0x31, &int1_src))
235  {
236  continue;
237  }
238 
239  if(self->_irq_flag)
240  {
241  self->_irq_flag = 0;
242 
243  if (self->_event_handler != NULL)
244  {
245  self->_event_handler(self, TWR_LIS2DH12_EVENT_ALARM, self->_event_param);
246  }
247  }
248  }
249 
250  self->_state = TWR_LIS2DH12_STATE_MEASURE;
251 
252  return;
253  }
254  default:
255  {
256  self->_state = TWR_LIS2DH12_STATE_ERROR;
257 
258  continue;
259  }
260  }
261  }
262 }

300 bool twr_lis2dh12_set_alarm(twr_lis2dh12_t *self, twr_lis2dh12_alarm_t *alarm)
301 {
302  if (alarm != NULL)
303  {
304  // Enable alarm
305  self->_alarm_active = true;
306 
307  // Disable IRQ first to change the registers
308  if (!twr_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, 0x30, 0x00))
309  {
310  return false;
311  }
312 
313  // Recalculate threshold to the 4g full-scale setting
314  uint8_t int1_ths = (uint8_t)((alarm->threshold) / 0.031f);
315 
316  // Ensure minimum threshold level
317  if (int1_ths == 0)
318  {
319  int1_ths = 1;
320  }
321 
322  if (!twr_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, 0x32, int1_ths))
323  {
324  return false;
325  }
326 
327  uint8_t int1_duration = (uint8_t)(alarm->duration / 1); // /10
328  if (!twr_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, 0x33, int1_duration))
329  {
330  return false;
331  }
332 
333  // CTRL_REG3
334  uint8_t ctrl_reg3 = (1 << 6);
335  if (!twr_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, 0x22, ctrl_reg3))
336  {
337  return false;
338  }
339 
340  // CTRL_REG6 - invert interrupt
341  uint8_t ctrl_reg6 = (1 << 1);
342  if (!twr_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, 0x25, ctrl_reg6))
343  {
344  return false;
345  }
346 
347  // ctr_reg5
348  uint8_t ctrl_reg5 = (0 << 3); // latch interrupt request
349  if (!twr_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, 0x24, ctrl_reg5))
350  {
351  return false;
352  }
353 
354  // INT_CFG1
355  uint8_t int_cfg1;
356 
357  int_cfg1 = 0; //(1 << 7) | (1 << 6); // AOI = 0, 6D = 1
358 
359  int_cfg1 |= (alarm->z_high) ? (1 << 5) : 0;
360  int_cfg1 |= (alarm->z_low) ? (1 << 4) : 0;
361 
362  int_cfg1 |= (alarm->y_high) ? (1 << 3) : 0;
363  int_cfg1 |= (alarm->y_low) ? (1 << 2) : 0;
364 
365  int_cfg1 |= (alarm->x_high) ? (1 << 1) : 0;
366  int_cfg1 |= (alarm->x_low) ? (1 << 0) : 0;
367 
368  if (!twr_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, 0x30, int_cfg1))
369  {
370  return false;
371  }
372 
373  twr_exti_register(TWR_EXTI_LINE_PB6, TWR_EXTI_EDGE_FALLING, _twr_lis2dh12_interrupt, self);
374  }
375  else
376  {
377  // Disable alarm
378  self->_alarm_active = false;
379 
380  if (!twr_i2c_memory_write_8b(self->_i2c_channel, self->_i2c_address, 0x30, 0x00))
381  {
382  return false;
383  }
384 
385  twr_exti_unregister(TWR_EXTI_LINE_PB6);
386  }
387 
388  twr_lis2dh12_measure(self);
389 
390  return true;
391 }
