144 void twr_module_lcd_draw_image(int left, int top, const twr_image_t *img)
145 {
146  uint8_t line;
147  uint8_t row;
148  uint8_t bytes_per_row = img->width / 8;
149 
150  if(img->width % 8 != 0)
151  {
152  bytes_per_row++;
153  }
154 
155  twr_tick_t start = twr_tick_get();
156 
157  for (row = 0; row < img->height; row++) {
158  for (line = 0; line < img->width; line++) {
159  uint32_t byte_offset = line / 8 + row * bytes_per_row;
160  uint32_t bit = line % 8;
161  twr_gfx_draw_pixel(&_twr_module_lcd.gfx, line + left, row + top, (img->data[byte_offset]) & (1 << bit));
162  }
163  }
164  volatile twr_tick_t duration = twr_tick_get() - start;
165  (void)duration;
166 
167 }

179 static void _twr_module_lcd_button_event_handler(twr_button_t *self, twr_button_event_t event, void *event_param)
180 {
181  (void) event_param;
182 
183  if (self == &_twr_module_lcd.button_left)
184  {
185  if (event == TWR_BUTTON_EVENT_PRESS)
186  {
187  _twr_module_lcd.event_handler(TWR_MODULE_LCD_EVENT_LEFT_PRESS, _twr_module_lcd.event_param);
188  }
189  if (event == TWR_BUTTON_EVENT_RELEASE)
190  {
191  _twr_module_lcd.event_handler(TWR_MODULE_LCD_EVENT_LEFT_RELEASE, _twr_module_lcd.event_param);
192  }
193  if (event == TWR_BUTTON_EVENT_CLICK)
194  {
195  _twr_module_lcd.event_handler(TWR_MODULE_LCD_EVENT_LEFT_CLICK, _twr_module_lcd.event_param);
196  }
197  if (event == TWR_BUTTON_EVENT_HOLD)
198  {
199  if (_twr_module_lcd.button_right._state)
200  {
201  _twr_module_lcd.event_handler(TWR_MODULE_LCD_EVENT_BOTH_HOLD, _twr_module_lcd.event_param);
202  // Force _hold_signalized to true, so the hold event of the second button won't trigger which would cause event duplication
203  _twr_module_lcd.button_right._hold_signalized = true;
204  }
205  else
206  {
207  _twr_module_lcd.event_handler(TWR_MODULE_LCD_EVENT_LEFT_HOLD, _twr_module_lcd.event_param);
208  }
209  }
210  }
211 
212  if (self == &_twr_module_lcd.button_right)
213  {
214  if (event == TWR_BUTTON_EVENT_PRESS)
215  {
216  _twr_module_lcd.event_handler(TWR_MODULE_LCD_EVENT_RIGHT_PRESS, _twr_module_lcd.event_param);
217  }
218  if (event == TWR_BUTTON_EVENT_RELEASE)
219  {
220  _twr_module_lcd.event_handler(TWR_MODULE_LCD_EVENT_RIGHT_RELEASE, _twr_module_lcd.event_param);
221  }
222  if (event == TWR_BUTTON_EVENT_CLICK)
223  {
224  _twr_module_lcd.event_handler(TWR_MODULE_LCD_EVENT_RIGHT_CLICK, _twr_module_lcd.event_param);
225  }
226  if (event == TWR_BUTTON_EVENT_HOLD)
227  {
228  if (_twr_module_lcd.button_left._state)
229  {
230  _twr_module_lcd.event_handler(TWR_MODULE_LCD_EVENT_BOTH_HOLD, _twr_module_lcd.event_param);
231  // Force _hold_signalized to true, so the hold event of the second button won't trigger which would cause event duplication
232  _twr_module_lcd.button_left._hold_signalized = true;
233  }
234  else

237  }
238  }
